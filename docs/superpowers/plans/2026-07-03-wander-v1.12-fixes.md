```ts
export function haversine(lat1: number, lng1: number, lat2: number, lng2: number): number {
  const R = 6371000
  const toRad = (deg: number) => (deg * Math.PI) / 180
  const dLat = toRad(lat2 - lat1)
  const dLng = toRad(lng2 - lng1)
  const a =
    Math.sin(dLat / 2) ** 2 +
    Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.sin(dLng / 2) ** 2
  return 2 * R * Math.asin(Math.min(1, Math.sqrt(a)))
}

export function bearing(lat1: number, lng1: number, lat2: number, lng2: number): number {
  const toRad = (d: number) => (d * Math.PI) / 180
  const dLng = toRad(lng2 - lng1)
  const y = Math.sin(dLng) * Math.cos(toRad(lat2))
  const x =
    Math.cos(toRad(lat1)) * Math.sin(toRad(lat2)) -
    Math.sin(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.cos(dLng)
  return ((Math.atan2(y, x) * 180) / Math.PI + 360) % 360
}
```

```ts
const REQUEST_LOCATION_WALL_CLOCK_MS = 15000

export function requestLocationViaNavigatorOrBridge(dispatch: (e: PhoneEvent) => void): void {
  if (!navigator.geolocation) {
    bridgeGeolocate().then((fix) => {
      if (fix) {
        dispatch({ type: 'location-acquired', lat: fix.lat, lng: fix.lng, source: 'bridge' })
        return
      }
      dispatch({ type: 'location-failed', message: 'Geolocation not supported on this device.' })
    })
    return
  }

  let settled = false
  const settle = (fn: () => void) => {
    if (settled) return
    settled = true
    fn()
  }

  const wallClock = setTimeout(() => {
    settle(() => {
      bridgeGeolocate().then((fix) => {
        if (fix) {
          dispatch({ type: 'location-acquired', lat: fix.lat, lng: fix.lng, source: 'bridge' })
          return
        }
        dispatch({ type: 'location-failed', message: 'Location request timed out. Check your location settings and try again.' })
      })
    })
  }, REQUEST_LOCATION_WALL_CLOCK_MS)

  navigator.geolocation.getCurrentPosition(
    (pos) => {
      settle(() => {
        clearTimeout(wallClock)
        dispatch({
          type: 'location-acquired',
          lat: pos.coords.latitude,
          lng: pos.coords.longitude,
          source: 'native',
        })
      })
    },
    (err) => {
      settle(() => {
        clearTimeout(wallClock)
        bridgeGeolocate().then((fix) => {
          if (fix) {
            dispatch({ type: 'location-acquired', lat: fix.lat, lng: fix.lng, source: 'bridge' })
            return
          }
          let message: string
          if (err.code === 1 /* PERMISSION_DENIED */) {
            message =
              'Location permission is required. If you\'ve already granted it, try force-quitting and reopening the app.'
          } else if (err.code === 2 /* POSITION_UNAVAILABLE */) {
            message = "Your location couldn't be determined. Make sure location services are enabled."
          } else {
            message = 'Location request timed out. Check your location settings and try again.'
          }
          dispatch({ type: 'location-failed', message })
        })
      })
    },
    { timeout: 10_000, maximumAge: 30_000 },
  )
}
```

```js
#!/usr/bin/env node
// Post-build pre-submission gate: every http/https URL literal in the
// BUILT bundle (dist/assets/*.js) must be covered by app.json's
// network.whitelist. src/__tests__/network-whitelist.test.ts only scans
// hand-authored source — this catches URLs that only appear after
// bundling (e.g. baked into a third-party dependency), which the
// EvenHub store's static scanner checks against the shipped bundle.
// See Wander_v2_Research.md L4.
import { readFileSync, readdirSync } from 'node:fs'
import { join } from 'node:path'

const ROOT = new URL('..', import.meta.url).pathname
const DIST_ASSETS = join(ROOT, 'dist', 'assets')

function readAppJsonWhitelist() {
  const raw = JSON.parse(readFileSync(join(ROOT, 'app.json'), 'utf8'))
  const networkPerm = raw.permissions?.find((p) => p.name === 'network')
  return networkPerm?.whitelist ?? []
}

function extractUrls(content) {
  const matches = content.match(/https?:\/\/[^\s'"`,)>]+/g) ?? []
  return [...new Set(matches)]
}

function isCovered(url, whitelist) {
  return whitelist.some((entry) => url.startsWith(entry))
}

const whitelist = readAppJsonWhitelist()
const files = readdirSync(DIST_ASSETS).filter((f) => f.endsWith('.js'))
const allUrls = files.flatMap((f) => extractUrls(readFileSync(join(DIST_ASSETS, f), 'utf8')))
const uncovered = [...new Set(allUrls)].filter((url) => !isCovered(url, whitelist))

if (uncovered.length > 0) {
  console.error('[check-bundle-whitelist] Uncovered URLs found in built bundle:')
  for (const url of uncovered) console.error(`  ${url}`)
  console.error('\nAdd these to app.json permissions[].whitelist, or confirm they are false positives, before packing.')
  process.exit(1)
}
console.log(`[check-bundle-whitelist] OK — ${allUrls.length} URL(s) found in dist/assets, all covered.`)
```

### Task 20: Fix "Open in Maps" / "Website" links failing on Android (user bug report)

**Background (from a real user bug report, 2026-07-04):** A user reported "Page Web non disponible" / `net::ERR_UNKNOWN_URL_SCHEME` when tapping "Open in Maps" on the phone's Nearby tab, on Android, and guessed it was a cookie-consent problem. It isn't. The failing URL was `intent://www.google.com/maps/search/?...#Intent;scheme=https;package=com.google.android.apps.maps;...;end` — an Android-only "Intent URL" that Google's own Maps webpage generates to hand off to the native Maps app. A real browser (Chrome, etc.) knows how to parse `intent://`; the EvenHub host's embedded WebView does not, so it fails outright instead of handing off.

`src/phone/tabs/NearbyTab.tsx`'s "Open in Maps" link (`mapsUrl(poi)` building `https://www.google.com/maps/search/?api=1&query=lat,lng`, at line ~319) and its "Website" link (line ~328) are both plain `<a href={...} target="_blank" rel="noreferrer">` — and `target="_blank"` alone does **not** escape this specific embedded WebView. We know this because the exact same problem was already hit and fixed once, on the glasses side: `src/glasses/effects.ts`'s `defaultOpenUrl(url: string): void` (used for the glasses "Open in Safari" action) has a comment reading "Field test 2026-04-25 confirmed that `_blank` opens inside EvenHub's overlay, which captures glasses input" — and its fix is to try `window.open(url, '_system', 'noopener,noreferrer')` first (a convention Cordova/Capacitor-style WebView wrappers honor to escape to the OS browser), falling back to `_blank` only if `_system` is rejected (returns a falsy window). This fix was never applied to the phone side.

**This task reuses that already-field-tested pattern rather than inventing a new one.** No structural change to anything else — the glasses' own "Open in Safari"/`open-url` effect, and Wander's own on-glasses turn-by-turn navigation (which never touches Google Maps at all), are completely unrelated to this fix and untouched by it.
